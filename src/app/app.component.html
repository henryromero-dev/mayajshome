MayaJs, is not an application itself, but a tool that allows you to design a database schema, create a REST API, and generate a complete project structure. 


class AppComponent implements OnInit {
    constructor(private mayaService: MayaService) {}
    public ngOnInit () {
        const res = this.mayaService.getAll('customers', 1, 10);
        console.log(res);
    }
}
